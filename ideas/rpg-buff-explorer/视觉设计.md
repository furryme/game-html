# 像素风视觉设计（整合版）

> 单文件 HTML 像素风 RPG，地城探险。Canvas 程序化绘制（无外部图片），480x270 逻辑分辨率，PIXEL=4。
> 混合架构：Canvas 负责游戏画面，HTML 负责 UI。

### 2.7 装备变体叠加方法

装备通过**像素叠加层**实现，而非重新画整组精灵：

```javascript
// 基础战士精灵（无装备）
const WAR_ID_F1 = [ /* 8x12 数组 */ ];

// 装备叠加层 — 只定义被覆盖的像素
const OVL_HELMET_IRON = {
  // 覆盖头盔区域像素，5 = steel 色
  pixels: [
    [0, 0, 5, 5, 5, 5, 0, 0],    // y=1
    [0, 5, 5, 5, 5, 5, 5, 0],    // y=2
    [0, 5, 0, 5, 5, 0, 5, 0],    // y=3 (眼窝镂空=0，保留底层眼睛)
  ],
  offsetY: 1,  // 从 y=1 开始叠加
};

// 运行时叠加
function applyOverlay(base, overlay) {
  const result = base.map(row => [...row]);
  for (let r = 0; r < overlay.pixels.length; r++) {
    const y = r + overlay.offsetY;
    for (let c = 0; c < overlay.pixels[r].length; c++) {
      const v = overlay.pixels[r][c];
      if (v !== 0) result[y][c] = v;  // 非零覆盖
    }
  }
  return result;
}
```

每种装备（头盔/铠甲/武器/盾牌）定义一个叠加层，运行时按需叠加到基础精灵上：
- 头盔叠加层：覆盖 y=1-4 头部区域
- 铠甲叠加层：覆盖 y=5-9 躯干区域
- 武器叠加层：覆盖右手区域 + 延伸
- 盾牌叠加层：覆盖左手区域

初始化时预生成所有装备组合的精灵（战士 x 3 头盔 x 2 铠甲 x 3 武器 = ~54 种组合，缓存到 offscreen canvas）。

function rotate90(arr) {
  const h = arr.length, w = arr[0].length;
  const result = Array.from({ length: w }, () => Array(h).fill(0));
  for (let y = 0; y < h; y++)
    for (let x = 0; x < w; x++)
      result[x][h - 1 - y] = arr[y][x];
  return result;
}

function overlay(a, b) {
  return a.map((row, y) => row.map((v, x) => v || b[y][x]));
}

### 3.4 地砖瓦片（生成函数）

```javascript
function makeFloorTile(checkerboard = true, palette = null) {
  const p = palette || { 1: PAL.dark, 2: PAL.midDark };
  const tile = Array.from({ length: 16 }, () => Array(16).fill(1));
  if (checkerboard) {
    for (let y = 0; y < 16; y += 8)
      for (let x = 0; x < 16; x += 8)
        for (let dy = 0; dy < 8; dy++)
          for (let dx = 0; dx < 8; dx++)
            tile[y + dy][x + dx] = 2;
  }
  return { pattern: tile, palette: p };
}
```

### 3.7 自动瓦片（Autotile）

将瓦片拆成中心 + 5 边 + 4 角共 9 个小片，运行时根据邻居组合：

```javascript
const AUTOTILE_PIECES = {
  center: CENTER_8x8,
  edge_n: EDGE_N_8x8, edge_s: EDGE_S_8x8,
  edge_e: EDGE_E_8x8, edge_w: EDGE_W_8x8,
  corner_ne: CORNER_NE_6x6, corner_nw: CORNER_NW_6x6,
  corner_se: CORNER_SE_6x6, corner_sw: CORNER_SW_6x6,
};

function buildAutotile(n, e, s, w) {
  const result = Array.from({ length: 16 }, () => Array(16).fill(0));
  // 叠加中心、边、角（取非零值）
  overlayAt(result, AUTOTILE_PIECES.center, 4, 4);
  if (n) overlayAt(result, AUTOTILE_PIECES.edge_n, 4, 0);
  if (e) overlayAt(result, AUTOTILE_PIECES.edge_e, 8, 4);
  if (s) overlayAt(result, AUTOTILE_PIECES.edge_s, 4, 8);
  if (w) overlayAt(result, AUTOTILE_PIECES.edge_w, 0, 4);
  if (n && e) overlayAt(result, AUTOTILE_PIECES.corner_ne, 8, 0);
  if (n && w) overlayAt(result, AUTOTILE_PIECES.corner_nw, 0, 0);
  if (s && e) overlayAt(result, AUTOTILE_PIECES.corner_se, 8, 8);
  if (s && w) overlayAt(result, AUTOTILE_PIECES.corner_sw, 0, 8);
  return result;
}
```

### 3.8 缓存策略

**初始化阶段**：
1. 加载调色板
2. 生成/旋转所有瓦片数组
3. autotile 预合成 21 种组合
4. 全部绘制到离屏 Canvas 纹理集
5. 记录每个瓦片在纹理集中的 (x, y) 坐标

**帧循环中**：从纹理集 `drawImage`，不再逐像素绘制。

```javascript
// 纹理集布局 256x256 实际像素
// 每个格子 64x64 (16x16 逻辑像素 x PIXEL=4)
const atlas = document.createElement('canvas');
atlas.width = 256; atlas.height = 256;
const atlasCtx = atlas.getContext('2d');
atlasCtx.imageSmoothingEnabled = false;

// 初始化时把每个瓦片画到纹理集对应位置
function buildAtlas(tiles, palette) {
  const slotSize = 64; // 16 * PIXEL
  let idx = 0;
  const positions = {};
  for (const [name, tile] of Object.entries(tiles)) {
    const col = idx % 4, row = Math.floor(idx / 4);
    drawSprite(atlasCtx, col * slotSize, row * slotSize, tile.pattern, palette);
    positions[name] = { x: col * slotSize, y: row * slotSize };
    idx++;
  }
  return { canvas: atlas, positions };
}
```

### 5.2 粒子系统（5 种粒子类型）

```javascript
const FX = {
  // ① burst: 圆形爆发
  burst(x, y, color, count, speed, size),

  // ② spray: 方向性喷射（技能特效）
  spray(x, y, angle, spread, color, count),

  // ③ shower: 持续下落（碎屑/余烬）
  shower(color, count, gravity),

  // ④ floatText: 浮动文字（伤害数字）
  floatText(x, y, text, color, fontSize),

  // ⑤ ring: 环形扩散（升级/防御）
  ring(x, y, color, radius, speed),
};
```

核心实现（从专家 4 方案提炼，适配 PIXEL=4）：

```javascript
const fxList = [];
const floatingTexts = [];
let shakeOffset = { x: 0, y: 0 };
let shakeRemaining = 0;
let flashAlpha = 0;
let flashColor = null;

function FX.burst(x, y, color, count = 10, speed = 3, size = PIXEL) {
  for (let i = 0; i < count; i++) {
    const angle = (Math.PI * 2 * i) / count + Math.random() * 0.3;
    const spd = speed * (0.6 + Math.random() * 0.8);
    fxList.push({
      x, y,
      vx: Math.cos(angle) * spd,
      vy: Math.sin(angle) * spd,
      life: 20 + Math.random() * 20,
      maxLife: 40,
      color, size,
    });
  }
}

function FX.floatText(x, y, text, color, fontSize = 16) {
  floatingTexts.push({ x, y, text, color, fontSize, life: 60, vy: -1.2 });
}

function FX.shake(intensity = 3, duration = 100) {
  shakeRemaining = duration;
  shakeOffset.max = intensity;
}

function FX.flash(color = '#ffffff', duration = 50, opacity = 0.3) {
  flashColor = color;
  flashAlpha = opacity;
  flashDuration = duration;
}

function FX.update(dt) {
  for (let i = fxList.length - 1; i >= 0; i--) {
    const p = fxList[i];
    p.x += p.vx; p.y += p.vy;
    p.vy += 0.05;
    p.life--;
    if (p.life <= 0) fxList.splice(i, 1);
  }
  for (let i = floatingTexts.length - 1; i >= 0; i--) {
    const t = floatingTexts[i];
    t.y += t.vy;
    t.life--;
    if (t.life <= 0) floatingTexts.splice(i, 1);
  }
  if (shakeRemaining > 0) {
    const max = shakeOffset.max;
    shakeOffset.x = (Math.random() - 0.5) * max * 2;
    shakeOffset.y = (Math.random() - 0.5) * max * 2;
    shakeRemaining -= dt;
  } else {
    shakeOffset.x = 0; shakeOffset.y = 0;
  }
  if (flashAlpha > 0) {
    flashAlpha *= 0.9;
    if (flashAlpha < 0.01) flashAlpha = 0;
  }
}

function FX.render(ctx) {
  ctx.save();
  ctx.translate(shakeOffset.x, shakeOffset.y);
  ctx.imageSmoothingEnabled = false;
  fxList.forEach(p => {
    ctx.globalAlpha = Math.max(0, p.life / p.maxLife);
    ctx.fillStyle = p.color;
    ctx.fillRect(
      Math.floor(p.x / PIXEL) * PIXEL,
      Math.floor(p.y / PIXEL) * PIXEL,
      p.size, p.size
    );
  });
  floatingTexts.forEach(t => {
    ctx.globalAlpha = Math.max(0, t.life / 60);
    ctx.fillStyle = t.color;
    ctx.font = `bold ${t.fontSize}px "Press Start 2P", monospace`;
    ctx.textAlign = 'center';
    ctx.fillText(t.text, t.x, t.y);
  });
  ctx.globalAlpha = 1;
  ctx.restore();
  if (flashAlpha > 0) {
    ctx.globalAlpha = flashAlpha;
    ctx.fillStyle = flashColor;
    ctx.fillRect(0, 0, ctx.canvas.width, ctx.canvas.height);
    ctx.globalAlpha = 1;
  }
}

function FX.clear() {
  fxList.length = 0;
  floatingTexts.length = 0;
  shakeOffset = { x: 0, y: 0 };
  shakeRemaining = 0;
  flashAlpha = 0;
}
```

```javascript
let starfield = [];

function Starfield.init(count = 40, colors = null) {
  starfield = [];
  colors = colors || [PAL.white, '#8888aa', '#6c5ce7'];
  for (let i = 0; i < count; i++) {
    starfield.push({
      x: Math.random(),
      y: Math.random(),
      size: Math.random() > 0.8 ? PIXEL : PIXEL / 2,
      color: colors[Math.floor(Math.random() * colors.length)],
      twinkle: Math.random() * Math.PI * 2,
      speed: 0.5 + Math.random() * 1.5,
      drift: (Math.random() - 0.5) * 0.0001,
    });
  }
}

function Starfield.render(ctx, w, h, time) {
  starfield.forEach(s => {
    const alpha = 0.2 + 0.3 * Math.sin(s.twinkle + time * s.speed * 0.001);
    ctx.globalAlpha = alpha;
    ctx.fillStyle = s.color;
    ctx.fillRect(
      Math.floor(s.x * w / PIXEL) * PIXEL,
      Math.floor(s.y * h / PIXEL) * PIXEL,
      s.size, s.size
    );
    s.y += s.drift * (time * 0.01);
    if (s.y > 1) s.y = 0;
    if (s.y < 0) s.y = 1;
  });
  ctx.globalAlpha = 1;
}

// 数字上滚：从旧值逐位滚动到新值
function animateNumber(element, from, to, duration = 300) {
  const start = performance.now();
  function tick(now) {
    const t = Math.min(1, (now - start) / duration);
    const val = Math.floor(from + (to - from) * t);
    element.textContent = val;
    if (t < 1) requestAnimationFrame(tick);
  }
  requestAnimationFrame(tick);
}

// ① 瓦片纹理集 — 初始化时生成一次
const tileAtlas = buildAtlas(dungeonTiles, FLOOR_THEMES[currentFloor]);

// ② 精灵预渲染 — 初始化时把每帧画到 offscreen canvas
const spriteCache = {};
function cacheSprite(key, frames, palette) {
  spriteCache[key] = frames.map(frame => {
    const c = document.createElement('canvas');
    const w = frame[0].length * PIXEL;
    const h = frame.length * PIXEL;
    c.width = w; c.height = h;
    const ctx = c.getContext('2d');
    drawSprite(ctx, 0, 0, frame, palette);
    return c;
  });
}

// ③ 背景 — 静态背景渲染到离屏 Canvas
const bgCanvas = document.createElement('canvas');
bgCanvas.width = 480; bgCanvas.height = 270;
```

帧循环中不再逐像素绘制，全部 `drawImage`。

### 6.3 帧循环结构

```javascript
let lastTime = 0;
function gameLoop(timestamp) {
  const dt = timestamp - lastTime;
  lastTime = timestamp;

  // 1. 更新
  if (!gameState.paused) {
    FX.update(dt);
    updateAnimations(dt);  // 精灵帧切换、环境动画
  }

  // 2. 渲染主 Canvas
  const ctx = mainCtx;
  ctx.clearRect(0, 0, 480, 270);
  ctx.save();
  ctx.translate(shakeOffset.x, shakeOffset.y);

  drawBackground(ctx, timestamp);      // 星尘
  drawTileMap(ctx, tileAtlas);         // 地图瓦片
  drawDecorations(ctx, timestamp);     // 火把/箱子
  drawEnemies(ctx, spriteCache);       // 敌人精灵
  drawPlayer(ctx, spriteCache);        // 玩家精灵

  ctx.restore();

  // 3. 渲染粒子 Canvas (overlay)
  FX.render(particlesCtx);

  // 4. 闪白
  if (flashAlpha > 0) {
    ctx.globalAlpha = flashAlpha;
    ctx.fillStyle = flashColor;
    ctx.fillRect(0, 0, 480, 270);
    ctx.globalAlpha = 1;
  }

  requestAnimationFrame(gameLoop);
}
```

### 6.4 性能保障

| 参数 | 目标值 | 说明 |
|------|--------|------|
| 游戏主循环 | 60fps (requestAnimationFrame) | 不额外设逻辑帧率 |
| 粒子上限 | 200 个 | 超限则 oldest 优先回收 |
| 浮动文字上限 | 10 个 | 超限丢弃最旧 |
| 星尘数量 | 30-50 个 | 用相对坐标，render 时映射 |
| 震屏最大强度 | 6px | Boss 死亡用 Lv.3 |
| 闪白最大时长 | 100ms | 超过影响操作 |

优化手段：
- 粒子用 `fillRect` 而非 `arc`（无路径计算）
- 所有坐标像素对齐（`Math.floor(x / PIXEL) * PIXEL`）
- 精灵帧预先渲染到 offscreen canvas，帧循环中 `drawImage`
- 瓦片纹理集初始化时预合成，帧循环中 `drawImage`
- 星尘背景渲染到 offscreen canvas，每 30 帧更新一次
- 粒子数组用 splice 反向遍历删除（无 GC 压力）

## 附：精灵绘制引擎最终版

```javascript
const PIXEL = 4;

/**
 * 在 ctx 上绘制一个二维数组精灵
 * @param {CanvasRenderingContext2D} ctx
 * @param {number} x - 左上角 X
 * @param {number} y - 左上角 Y
 * @param {number[][]} pattern - 二维数组，0=透明
 * @param {Object} palette - { 1: '#hex', 2: '#hex', ... }
 * @param {Object} opts - { flip?, alpha?, scale? }
 */
function drawSprite(ctx, x, y, pattern, palette, opts = {}) {
  const { flip = false, alpha = 1, scale = 1 } = opts;
  const p = PIXEL * scale;
  ctx.globalAlpha = alpha;

  if (flip) {
    ctx.save();
    ctx.translate(x + pattern[0].length * p, y);
    ctx.scale(-1, 1);
    x = 0;
  }

  for (let row = 0; row < pattern.length; row++) {
    for (let col = 0; col < pattern[row].length; col++) {
      const idx = pattern[row][col];
      if (!idx) continue;
      // 半透明像素: 0 < idx < 1
      if (idx < 1) {
        ctx.globalAlpha = alpha * idx;
        ctx.fillStyle = palette['_alpha'] || '#a0b8d8';
      } else {
        ctx.fillStyle = palette[idx];
      }
      ctx.fillRect(x + col * p, y + row * p, p, p);
    }
  }

  if (flip) ctx.restore();
  ctx.globalAlpha = 1;
}

/** 像素对齐的 fillRect */
function pixelRect(ctx, x, y, w, h, color) {
  ctx.fillStyle = color;
  ctx.fillRect(
    Math.floor(x / PIXEL) * PIXEL,
    Math.floor(y / PIXEL) * PIXEL,
    w * PIXEL,
    h * PIXEL
  );
}
```
